68void BuzzerPlayRtttl(const char * rtttl_melody){
69 uint8_t default_dur = 4;
70 uint8_t default_oct = 6;
71 int bpm = 63;
72 int num;
73 long wholenote;
74 long duration;
75 uint8_t note;
76 uint8_t scale;
77
78 /* find the start (skip name, etc) */
79 while(*rtttl_melody != ':') rtttl_melody++; // ignore name
80 rtttl_melody++; // skip ':'
81
82 /* get default duration */
83 if(*rtttl_melody == 'd'){
84 rtttl_melody++;
85 rtttl_melody++; // skip "d="
86 num = 0;
87 while(isDigit(*rtttl_melody)){
88 num = (num * 10) + (*rtttl_melody++ - '0');
89 }
90 if(num > 0) default_dur = num;
91 rtttl_melody++; // skip comma
92 }
93
94 /* get default octave */
95 if(*rtttl_melody == 'o'){
96 rtttl_melody++;
97 rtttl_melody++; // skip "o="
98 num = *rtttl_melody++ - '0';
99 if(num >= 3 && num <=7) default_oct = num;
100 rtttl_melody++; // skip comma
101 }
102
103 /* get BPM */
104 if(*rtttl_melody == 'b'){
105 rtttl_melody++;
106 rtttl_melody++; // skip "b="
107 num = 0;
108 while(isDigit(*rtttl_melody)){
109 num = (num * 10) + (*rtttl_melody++ - '0');
110 }
111 bpm = num;
112 rtttl_melody++; // skip colon
113 }
114
115 /* BPM usually expresses the number of quarter notes per minute */
116 wholenote = (60 * 1000L / bpm) * 4; // this is the time for whole note (in milliseconds)
117
118 /* now begin note loop */
119 while(*rtttl_melody){
120 /* first, get note duration, if available */
121 num = 0;
122 while(isDigit(*rtttl_melody)){
123 num = (num * 10) + (*rtttl_melody++ - '0');
124 }
125 if(num){
126 duration = wholenote / num;
127 }else{
128 duration = wholenote / default_dur; // we will need to check if we are a dotted note after
129 }
130 /* now get the note */
131 note = 0;
132 switch(*rtttl_melody){
133 case 'c':
134 note = 1;
135 break;
136 case 'd':
137 note = 3;
138 break;
139 case 'e':
140 note = 5;
141 break;
142 case 'f':
143 note = 6;
144 break;
145 case 'g':
146 note = 8;
147 break;
148 case 'a':
149 note = 10;
150 break;
151 case 'b':
152 note = 12;
153 break;
154 case 'p':
155 default:
156 note = 0;
157 }
158 rtttl_melody++;
159 /* now, get optional '#' sharp */
160 if(*rtttl_melody == '#'){
161 note++;
162 rtttl_melody++;
163 }
164 /* now, get optional '.' dotted note */
165 if(*rtttl_melody == '.'){
166 duration += duration/2;
167 rtttl_melody++;
168 }
169 /* now, get scale */
170 if(isDigit(*rtttl_melody)){
171 scale = *rtttl_melody - '0';
172 rtttl_melody++;
173 }else{
174 scale = default_oct;
175 }
176 scale += OCTAVE_OFFSET;
177
178 if(*rtttl_melody == ','){
179 rtttl_melody++; // skip comma for next note (or we may be at the end)
180 }
181 /* now play the note */
182 if(note){
183 BuzzerPlayTone(notes[(scale - 4) * 12 + note], duration);
184 }
185 else{
186 DelayMs(duration);
187 }
188 }
189}
